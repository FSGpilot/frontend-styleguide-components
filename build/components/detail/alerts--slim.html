{# 
  @param classes: usa-alert-info, usa-alert-success, usa-alert-warning, usa-alert-error, usa-alert-paragraph, usa-alert-slim. 
  @param role: e.g. 'alert'. 
  @param title: String, title of the alert. 
  @param content: String, the text content of the alert. 

  @usage: 
    {% import "alerts/alerts.njk" as alerts %}

  @return: the HTML for an alert-box.
#}
